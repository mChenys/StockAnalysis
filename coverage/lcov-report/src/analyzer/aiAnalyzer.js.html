    /**
     * 执行完整的AI股票分析
     */
    async analyzeStock(symbol, analysisType = 'technical', options = {}) {
        try {
            logger.info(`Starting AI analysis for ${symbol} - ${analysisType}`);
 
            // 1. 获取实时股票数据
            const stockData = await this.getStockData(symbol);
            
            // 2. 根据分析类型准备数据
            const analysisData = await this.prepareAnalysisData(symbol, analysisType, stockData);
            
            // 3. 调用AI进行分析
            const aiResult = await this.performAIAnalysis(symbol, analysisType, analysisData);
            
            // 4. 保存分析结果
            const result = {
                symbol,
                analysisType,
                analysis: aiResult,
                rawData: stockData,
                timestamp: new Date(),
                model: this.selectBestModel(analysisType)
            };
            
            await this.saveAnalysis(result);
            
            // 5. 如果是重要信号，发送推送
            if (options.pushNotification !== false && this.shouldPushNotification(result)) {
                await this.pushAnalysisResult(result);
            }
            
            logger.info(`AI analysis completed for ${symbol}`);
            return result;
            
        } catch (error) {
            logger.error(`AI analysis failed for ${symbol}:`, error);
            throw error;
        }
    }
 
    /**
     * 获取股票数据
     */
    async getStockData(symbol) {
        try {
            // 获取实时价格
            const realTimeData = await stockCrawler.getRealTimePrice(symbol);
            
            // 获取历史数据
            const historicalData = await stockCrawler.getHistoricalData(symbol, '3m');
            
            // 获取技术指标
            const technicalIndicators = await stockCrawler.getTechnicalIndicators(symbol);
            
            return {
                ...realTimeData,
                historical: historicalData,
                indicators: technicalIndicators
            };
            
        } catch (error) {
            logger.error(`Failed to get stock data for ${symbol}:`, error);
            // 返回模拟数据以确保系统可以继续运行
            return this.getMockStockData(symbol);
        }
    }
 
    /**

    /**
     * 执行AI分析
     */
    async performAIAnalysis(symbol, analysisType, data) {
        const template = this.analysisTemplates[analysisType];
        if (!template) {
            throw new Error(`Unknown analysis type: ${analysisType}`);
        }
 
        // 填充模板
        const prompt = this.fillTemplate(template, data);
        
        // 选择AI模型
        const modelName = this.selectBestModel(analysisType);
        if (!modelName) {
            throw new Error('No available AI model for analysis');
        }
        
        // 调用AI
        const analysis = await modelManager.callModel(modelName, prompt);
        return analysis;
    }
 
    /**
     * 填充模板
     */
    fillTemplate(template, data) {
        let filled = template;
        Object.keys(data).forEach(key => {
            const value = data[key] !== null && data[key] !== undefined ? data[key] : 'N/A';
            filled = filled.replace(new RegExp(`{${key}}`, 'g'), value);
        });
        return filled;
    }
 
    /**
     * 选择最适合的AI模型
     */
    selectBestModel(analysisType) {
        const models = modelManager.getModels();
        const activeModels = models.filter(m => m.active);
        
        if (activeModels.length === 0) return null;
        
        // 根据分析类型选择最合适的模型
        const preferences = {
            technical: ['gpt-4', 'claude-3', 'gemini-pro'],
            fundamental: ['gpt-4', 'claude-3', 'gemini-pro'],
            sentiment: ['gpt-3.5-turbo', 'gemini-pro'],
            risk: ['claude-3', 'gpt-4']
        };
        
        const preferred = preferences[analysisType] || ['gpt-3.5-turbo'];
        
        for (const pref of preferred) {
            const model = activeModels.find(m => 
                m.model.toLowerCase().includes(pref.split('-')[0]) || 
                m.name.toLowerCase().includes(pref)
            );
            if (model) return model.name;
        }
        
        return activeModels[0].name;
    }
 
    /**
     * 判断是否应该推送通知
     */
    shouldPushNotification(analysisResult) {
        const analysis = analysisResult.analysis.toLowerCase();
        
        // 检测关键信号词
        const buySignals = ['强烈推荐买入', '买入信号', '突破', '金叉'];
        const sellSignals = ['卖出信号', '跌破', '死叉', '风险较大'];
        
        const hasBuySignal = buySignals.some(signal => analysis.includes(signal));
        const hasSellSignal = sellSignals.some(signal => analysis.includes(signal));
        
        return hasBuySignal || hasSellSignal;
    }

    /**
     * 保存分析结果
     */
    async saveAnalysis(result) {
        try {
            // 这里可以保存到数据库
            // 目前使用内存存储
            if (!global.analysisResults) {
                global.analysisResults = [];
            }
            
            global.analysisResults.unshift(result);
            
            // 只保留最近100条记录
            if (global.analysisResults.length > 100) {
                global.analysisResults = global.analysisResults.slice(0, 100);
            }
            
        } catch (error) {
            logger.error('Failed to save analysis:', error);
        }
    }
 
    /**
     * 获取分析历史
     */
    async getAnalysisHistory(symbol, analysisType = null, limit = 10) {
        if (!global.analysisResults) return [];
        
        let results = global.analysisResults.filter(r => r.symbol === symbol);
        
        if (analysisType) {
            results = results.filter(r => r.analysisType === analysisType);
        }
        
        return results.slice(0, limit);
    }

        return `$${marketCap}`;
    }
